# 題目
a013. 羅馬數字

# 解題思路
本題的核心在於實現羅馬數字與十進位整數之間的雙向轉換。
1. **羅馬數字轉整數**：利用字典建立字元與數值的映射。由左至右遍歷字串，若當前字元代表的值小於其右側字元的值，表示觸發減法規則（如 IV = 4），應扣除該值；反之則累加。
2. **計算差值**：將讀入的兩個羅馬數字分別轉成整數後，計算兩者相減的絕對值 `abs()`。
3. **整數轉回羅馬數字**：建立一個由大到小的數值與羅馬符號組合清單（包含 CM、CD、XL 等複合符號）。透過貪婪演算法（Greedy）由大至小不斷扣除對應數值並拼接符號。特別注意，若差值為 0，則依題目規定直接輸出 "ZERO"。

18081907

# 程式碼
```python
import sys

def roman_to_int(s):
    roman_map = {'I': 1, 'V': 5, 'X': 10, 'L': 50, 'C': 100, 'D': 500, 'M': 1000}
    total = 0
    for i in range(len(s)):
        if i + 1 < len(s) and roman_map[s[i]] < roman_map[s[i+1]]:
            total -= roman_map[s[i]]
        else:
            total += roman_map[s[i]]
    return total

def int_to_roman(num):
    if num == 0:
        return "ZERO"
    val_map = [
        (1000, 'M'), (900, 'CM'), (500, 'D'), (400, 'CD'),
        (100, 'C'), (90, 'XC'), (50, 'L'), (40, 'XL'),
        (10, 'X'), (9, 'IX'), (5, 'V'), (4, 'IV'), (1, 'I')
    ]
    result = []
    for value, symbol in val_map:
        while num >= value:
            result.append(symbol)
            num -= value
    return "".join(result)

def main():
    for line in sys.stdin:
        line = line.strip()
        if line == "#" or not line:
            break
        parts = line.split()
        if len(parts) < 2:
            continue
        num1 = roman_to_int(parts[0])
        num2 = roman_to_int(parts[1])
        print(int_to_roman(abs(num1 - num2)))

if __name__ == '__main__':
    main()
