# 題目
a020. 身分證字號檢查

# 解題思路
本題旨在實作台灣身分證字號的驗證演算法。步驟如下：
1. **字母轉換**：建立字典將開頭的英文字母映射至對應的兩位數代號。將代號拆解為十位數與個位數後，進行初步加權計算（十位數 $\times 1$ + 個位數 $\times 9$）。
2. **數值加權**：身分證字號第 2 至 9 碼的數字，依序乘以權重 8 至 1 並累加至總和中。
3. **檢查碼累加**：最後一碼（第 10 碼）乘上權重 1 後併入總和。
最後，判斷總和是否能被 10 整除。若餘數為 0 則輸出 "real"，否則輸出 "fake"。

# 解題編號
18087635

# 程式碼
```python
import sys

def solve():
    letter_map = {
        'A': 10, 'B': 11, 'C': 12, 'D': 13, 'E': 14, 'F': 15, 'G': 16, 'H': 17,
        'I': 34, 'J': 18, 'K': 19, 'L': 20, 'M': 21, 'N': 22, 'O': 35, 'P': 23,
        'Q': 24, 'R': 25, 'S': 26, 'T': 27, 'U': 28, 'V': 29, 'W': 32, 'X': 30,
        'Y': 31, 'Z': 33
    }
    
    for line in sys.stdin:
        id_str = line.strip()
        if not id_str:
            continue
            
        first_letter = id_str[0].upper()
        num = letter_map[first_letter]
        
        n1 = num // 10
        n2 = num % 10
        total = n1 * 1 + n2 * 9
        
        weights = [8, 7, 6, 5, 4, 3, 2, 1]
        for i in range(1, 9):
            total += int(id_str[i]) * weights[i-1]
            
        total += int(id_str[9]) * 1
        
        if total % 10 == 0:
            print("real")
        else:
            print("fake")

if __name__ == '__main__':
    solve()
