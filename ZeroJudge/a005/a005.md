# 題目
a005. Eva 的回家作業

# 解題思路
讀取輸入的數列前四項 $a, b, c, d$。透過檢查 $b - a == c - b$ 來判斷其是否為等差數列。若是等差數列，則第五項為 $d + (b - a)$；若否，則判定為等比數列，第五項為 $d \times (b / a)$。計算出第五項後，將前五項數值以空白分隔依序輸出。

# 解題編號
18077547

# 程式碼
```python
import sys

def solve():
    lines = [line.strip() for line in sys.stdin if line.strip()]
    if not lines:
        return
    for line in lines[1:]:
        nums = [int(x) for x in line.split()]
        a, b, c, d = nums[0], nums[1], nums[2], nums[3]
        if (b - a) == (c - b):
            diff = b - a
            e = d + diff
        else:
            ratio = b // a
            e = d * ratio
        nums.append(e)
        print(*nums)

if __name__ == '__main__':
    solve()
